bahrein_decision = {
	is_shown = {
		this ?= c:PER
		NOT = { has_variable = bahrain_var }
		country_has_primary_culture = cu:persian
		is_subject = no
		exists = c:BHN
		c:BHN = {
			NOT = { is_subject_of = ROOT }
		}
	}

	possible = {
				c:PER = {
			navy_size >= 8
		}
		
		c:BHN = {
			relations:root >= relations_threshold:cordial
		}
	}

	when_taken = {
		s:STATE_ABU_DHABI = {
				add_claim = ROOT
		}
		set_variable = bahrain_var
		custom_tooltip = bahrein_event_tt
		hidden_effect = {
			trigger_event = {
				id = iran.1
				days = 0
				popup = yes
			}
		}

		#will get the oman treaty port, because of the low prestige of oman you cant get the treaty port inside the diplomatic playXDDD
		s:STATE_BALUCHISTAN= {
			set_owner_of_provinces = { 
				country = ROOT
				provinces = {xB7B2BD}
			}
			state_type = treaty_port
		}

		create_diplomatic_play = {
			name = push_back_oman

			target_state = s:STATE_ABU_DHABI.region_state:OMA
			
			war = yes
			
			type = dp_return_state

			add_war_goal = {
				holder = c:PER
				type = return_state
				target_state = s:STATE_LARISTAN.region_state:OMA
			}
			
			add_war_goal = {
				holder = c:OMA
				type = conquer_state
				target_state = s:STATE_LARISTAN.region_state:PER
			}
		}
	}

	ai_chance = {
		base = 10
	}
}

turan_decision = {
	is_shown = {
		NOT = { has_variable = turan_var }
		country_has_primary_culture = cu:persian
		is_subject = no
		exists = c:KHI
		c:KHI = {
			NOT = { is_subject_of = ROOT }
		}
	}

	possible = {
		c:PER = {
			army_size >= 80
			gdp >= 8000000
			year <= 1880
		}
	}

	when_taken = {
		set_variable = turan_var
		custom_tooltip = turan_event_tt

		c:PER = {
			create_diplomatic_play = {
				name = turan_war
		
				target_state = s:STATE_KHIVA.region_state:KHI
				
				war = yes
			
				type = dp_annex_war



				add_war_goal = {
				holder = c:PER
				type = annex_country
				target_state = s:STATE_TAJIKISTAN.region_state:KOK
				}

				add_war_goal = {
				holder = c:PER
				type = annex_country
				target_state = s:STATE_UZBEKIA.region_state:BUK
				}


				add_war_goal = {
					holder = c:KHI
					type = conquer_state
					target_state = s:STATE_KHORASAN.region_state:PER
				}
				add_war_goal = {
					holder = c:BUK
					type = war_reparations
					target_state = s:STATE_KHORASAN.region_state:PER
				}
				add_war_goal = {
					holder = c:KOK
					type = war_reparations
					target_state = s:STATE_KHORASAN.region_state:PER
				}
				add_target_backers = {
					c:KOK
					c:BUK
				}

			}
		}
	}

	ai_chance = {
		base = 10
	}
}

		}

			trigger_event = {
				id = iran.1

		s:STATE_BALUCHISTAN= {
			set_owner_of_provinces = {

		}

		name = treaty_name_farmaan_e_hormuzd

		second_country = c:OMA

		is_draft = no

			{
				article = treaty_port

					{ state = s:STATE_LARISTAN.region_state:OMA }

				inputs = {

			}
		}
		}

primary_culturing_abaxtaran_decision = {
	is_shown = {
		country_has_primary_culture = cu:iranian
		NOT = {
			has_variable = primary_culturing_abaxtaran_var
		}
	}
	possible = {
		country_has_state_religion = rel:mazdayasna
		owns_entire_state_region = STATE_AZERBAIJAN
		owns_entire_state_region = STATE_ARMENIA
		owns_entire_state_region = STATE_KARS
		owns_entire_state_region = STATE_GREATER_CAUCASUS
		owns_entire_state_region = STATE_DAGESTAN
		owns_entire_state_region = STATE_NORTH_CAUCASUS
		owns_entire_state_region = STATE_KUBAN
	}
	when_taken = {
		add_primary_culture = cu:armenian
		add_primary_culture = cu:georgian
		add_primary_culture = cu:north_caucasian
		add_primary_culture = cu:circassian
		add_primary_culture = cu:chechen
		set_variable = primary_culturing_abaxtaran_var
		add_modifier = {
			name = re_integrate_cultures
			months = 36
		}
	}
	ai_chance = {
		base = 100
	}
}


primary_culturing_nemroz_decision = {
	is_shown = {
		this ?= c:PER
		NOT = {
			has_variable = primary_culturing_nemroz_var
		}
	}
	possible = {
		owns_entire_state_region = STATE_PASHTUNISTAN
		custom_tooltip = {
			text = has_primary_culturing_abaxtaran_var
			has_variable = primary_culturing_abaxtaran_var
		}
	}
	when_taken = {
		add_primary_culture = cu:pashtun
		add_primary_culture = cu:ashkenazi
		set_variable = primary_culturing_nemroz_var
		add_modifier = {
			name = re_integrate_cultures
			months = 24
		}
	}
	ai_chance = {
		base = 100
	}
}
